#The Palaiologan Renaissance
country_event = {
	id = byz_events.1
	title = byz_events.1.t
	desc = byz_events.1.d
	picture = ARTWORK_eventPicture
	
	fire_only_once = yes
	
	trigger = {
		tag = BYZ
		government = monarchy
		dynasty = "Palaiologos"
		NOT = { has_country_modifier = palaiologan_renaissance }
		NOT = { is_year = 1475 }
	}
	
	mean_time_to_happen = { 
		months = 12 #Adjusted from 50 days
	}
	#Adjusted options to require money to grant modifier, with an option to not finance it.
	option = { #Support the freethinkers!
		name = byz_events.1.a
		add_years_of_income = -2
		add_country_modifier = {
			name = palaiologan_renaissance
			duration = 18250
		}
	}
	option = { #Do not
		name = byz_events.1.b
		add_legitimacy = -5
	}
}

#Prince Orhan 1
country_event = {
    id = byz_events.28
	title = byz_events.28.t
	desc = byz_events.28.d
	picture = CIVIL_WAR_eventPicture
    
    fire_only_once = yes
    
    trigger = {
        tag = BYZ
        is_year = 1451
		NOT = { is_year = 1475 }
        TUR = {
            owns = 149
        }
        NOT = { war_with = TUR }
    }
    
    mean_time_to_happen = {
        days = 10
    }
    
    option = {
        name = byz_events.28.a #Release him!
        ai_chance = { factor = 20 }
        TUR = {
            country_event = { id = byz_events.29 }
            add_claim = 151
		}
        if = {
            limit = {
                TUR = {
                    owns = 149
                }
            }
            149 = {
                spawn_rebels = {
                    type = pretender_rebels
                    size = 2
                    leader = "Orhan Osmanoglu"
                    friend = BYZ
                }
            }
        }
        if = {
            limit = {
                TUR = {
                    NOT = { owns = 149 }
                }
            }
            TUR = {
                random_owned_province = {
                    limit = {
                        is_capital = no
                    }
                    spawn_rebels = {
                        type = pretender_rebels
                        size = 2
                        leader = "Orhan Osmanoglu"
                        friend = BYZ
                    }
                }
            }
        }
        clr_country_flag = subject_of_ottomans
		set_country_flag = defied_the_sultan
    }
    option = {
        name = byz_events.28.b #Threaten to release
        ai_chance = { factor = 60 }
        TUR = {
			country_event = { id = byz_events.30 }
		}
		clr_country_flag = subject_of_ottomans
		set_country_flag = defied_the_sultan
        add_prestige = 10
    }
    option = {
        name = byz_events.28.c #take him in as an advisor
        ai_chance = { factor = 20 }
        define_advisor = {
            type = recruitmaster
            name = "Sehzade Orhan Çelebi"
            location = 151
            skill = 1
        }
    }
}
