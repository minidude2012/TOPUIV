#Beyond Typus's colonial missions, hopefully both historical and dynamic.
#Note the missions for a first settlement in the Lesser Antilles, Eastern United States, and Canada are located in the vanilla 'Colonial_Missions' file

#Beyond Typus Colonial Region missions - Newfoundland
bt_establish_newfoundland_colony = {

	type = country
	
	category = DIP
	ai_mission = yes
	
	allow = {
		num_of_ports = 1
		has_idea = quest_for_the_new_world 
		religion_group = christian
		is_colonial_nation = no	
		colonial_newfoundland = {
			is_empty = yes
			range = ROOT
			has_discovered = ROOT
			OR = {
				ROOT = { NOT = { religion = catholic } }
				is_religion_grant_colonial_claim = no
				is_religion_grant_colonial_claim = ROOT
			}
		}
		NOT = {
			colonial_newfoundland = {
				owned_by = ROOT
			}
		}
		NOT = {
			colonial_newfoundland = {
				owner = {
					is_subject_of = ROOT
				}
			}
		}
		NOT = {
			has_country_modifier = "colonial_enthusiasm" 
		}
	}
	abort = {
		OR = {
			NOT = {
				colonial_newfoundland = {
					is_empty = yes
					has_discovered = ROOT
				}
			}
			NOT = { num_of_ports = 1 }
			NOT = { has_idea = quest_for_the_new_world }
		}
	}
	success = {
		colonial_newfoundland = {
			owned_by =  ROOT
			colonysize = 400
		}
	}
	immediate = {
		if = {
			limit = { ai = yes }
			random_owned_province = {
				limit = { has_port = yes }
				province_event = { 
					id = BeyondTypusHeavyHandedAIColonialEvents.4
					days = 10
					random = 50
				}
			}
		}
	}
	chance = {
		factor = 500
		modifier = {
			factor = 1.5
			OR = {
				personality = ai_colonialist
				AND = {
					has_idea_group = exploration_ideas
					has_idea_group = expansion_ideas
				}
				ruler_has_personality = expansionist_personality
				advisor = colonial_governor
				advisor = navigator
			}
		}
		modifier = {
			factor = 4.0
			OR = {
				tag = ENG
				tag = GBR
				tag = FRA
				tag = NAV
				tag = IRE
				tag = TNA
			}
		}	#High chance historical tag favouritism
		modifier = {
			factor = 2.0
			OR = {
				tag = HOL
				tag = NED
				tag = POR
				tag = CAS
				tag = SPA
				tag = BRI
			}
		}	#Low chance historical tag favouritism
		modifier = {
			factor = 2.0
			OR = {
				239 = {			#Bristol - Irish sea
					trade_share = {
						country = ROOT
						share = 25
					}
				}
				174 = {			#Bordeaux - Bordeaux
					trade_share = {
						country = ROOT
						share = 25
					}
				}
			}
		}	#Controls at least 25 percent of an immediately downsteam node
		modifier = {
			factor = 0.1
			NOT = {
				239 = {			#Bristol - Irish sea
					trade_share = {
						country = ROOT
						share = 10
					}
				}
				174 = {			#Bordeaux - Bordeaux
					trade_share = {
						country = ROOT
						share = 10
					}
				}
			}
		}	#Does not control at least 10 percent of an immediately downsteam node
		modifier = {
			factor = 0.25
			OR = {
				any_subject_country = {
					colonial_newfoundland = { owned_by = PREV }
				}
				any_ally = {
					colonial_newfoundland = { owned_by = PREV }
				}
			}
		}	#Very unlikely if a vassal or ally already owns land in the region
		modifier = {
			factor = 0
			is_subject_other_than_tributary_trigger = yes
			NOT = { liberty_desire = 50 }
			overlord = {
				colonial_newfoundland = { owned_by = PREV }
			}
		}	#Impossible if a content vassal and overlord has land in the region
	}
	effect = {
		if = {
			limit = {
				NOT = {
					AND = {
						full_idea_group = exploration_ideas
						full_idea_group = expansion_ideas
					}
				#	tag = CAS
				#	tag = SPA
				}
			}
			add_country_modifier = {
				name = "colonial_enthusiasm"
				duration = 1875
			}
		}
		if = {
			limit = {
				#OR = {
				AND = {
					full_idea_group = exploration_ideas
					full_idea_group = expansion_ideas
				}
				#	tag = CAS
				#	tag = SPA
				#}
			}
			add_country_modifier = {
				name = "bt_spanish_colonial_enthusiasm"
				duration = 625
			}
		}
	}
}
