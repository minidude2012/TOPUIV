recover_greece = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	target_areas_list = {
		morea_area
		western_aegean_area
		northern_greece_area
	}
	
	target_provinces = {
		NOT = { owned_by = ROOT }
		NOT = { owner = { alliance_with = ROOT } }
		NOT = { owner = { is_subject_of = ROOT } }
		NOT = { owner = { overlord_of = ROOT } }
	}
	allow = {
		normal_or_historical_nations = yes
		tag = BYZ
		is_free_or_tributary_trigger = yes
		NOT = { has_country_modifier = byzantium_resurgent }
		NOT = { has_country_flag = retook_greece_byz }
		is_at_war = no
		owns_core_province = 151
		any_province = {
			OR = {
				area = morea_area
				area = western_aegean_area
				area = northern_greece_area
			}
			NOT = { owned_by = ROOT }
			NOT = { owner = { alliance_with = ROOT } }
			NOT = { owner = { is_subject_of = ROOT } }
			NOT = { owner = { overlord_of = ROOT } }
		}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 5000			#Priority 1
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			mil = 6
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			if = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
				}
				remove_claim = ROOT
			}
		}
	}
	effect = {
		add_prestige = 10
		set_country_flag = retook_greece_byz
		add_country_modifier = {
			name = "byzantium_resurgent"
			duration = 3650
		}
	}
}
