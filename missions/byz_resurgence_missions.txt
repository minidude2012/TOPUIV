byz_reconquer_mainland_greece = { #15000
	
	type = country

	category = MIL
	ai_mission = yes
	
	target_provinces = {
		OR = {
			area = western_aegean_area
			area = northern_greece_area
			area = morea_area
			area = macedonia_area
			area = thrace_area
		}
		NOT = { owned_by = ROOT }
		NOT = { owner = { alliance_with = ROOT } }
		NOT = { owner = { is_subject_of = ROOT } }
		NOT = { owner = { overlord_of = ROOT } }
	}
	allow = {
		normal_or_historical_nations = yes
		OR = {
			tag = BYZ
			AND = {
				tag = ROM
				culture_group = byzantine
			}
		}
		is_free_or_tributary_trigger = yes
		NOT = { has_country_flag = byz_took_mainland_greece }
		NOT = { has_country_modifier = byz_greek_ambition }
		owns_core_province = 151
		any_province = {
			OR = {
				area = western_aegean_area
				area = northern_greece_area
				area = morea_area
				area = macedonia_area
				area = thrace_area
			}
			NOT = { owned_by = ROOT }
			NOT = { owner = { alliance_with = ROOT } }
			NOT = { owner = { is_subject_of = ROOT } }
			NOT = { owner = { overlord_of = ROOT } }
		}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		all_target_province = {
			owned_by = ROOT
		}
	}
	chance = {
		factor = 15000			#After Turn the Tide in BtPL @ 20 000
		modifier = {
			factor = 2
			mil = 4
		}
		modifier = {
			factor = 2
			mil = 6
		}
	}
	immediate = {
		every_target_province = {
			add_claim = ROOT
		}
	}
	abort_effect = {
		every_target_province = {
			if = {
				limit = {
					NOT = {
						is_permanent_claim = ROOT
					}
				}
				remove_claim = ROOT
			}
		}
	}
	effect = {
		add_prestige = 10
		set_country_flag = byz_took_mainland_greece
		add_country_modifier = {
			name = "byz_greek_ambition"
			duration = 1825
		}
	}
}

	type = country

	category = DIP
	ai_mission = yes
	
	allow = {
		normal_or_historical_nations = yes
		NOT = { has_country_modifier = military_vassalization }
		tag = BYZ
		exists = BUL
		is_free_or_tributary_trigger = yes
		BUL = {
			is_free_or_tributary_trigger = yes
			is_neighbor_of = ROOT
			NOT = { num_of_cities = ROOT }
			NOT = { total_development = 100 }
			NOT = { government = imperial_city }
		}
	}
	abort = {
		OR = {
			is_subject_other_than_tributary_trigger = yes
			NOT = { exists = BUL }
			BUL = { is_subject_other_than_tributary_trigger = yes }
			NOT = {
				has_casus_belli = {
					type = cb_vassalize_mission
					target = BUL
				}
			}
		}
	}
	success = {
		BUL = { vassal_of = BYZ }
	}
	chance = {
		factor = 2000
		modifier = {
			factor = 2
			NOT = { has_opinion = { who = BUL value = 0 } }
		}
	}
	immediate = {
		add_casus_belli = {
			type = cb_vassalize_mission
			months = 300
			target = BUL
		}
	}
	abort_effect = {
		remove_casus_belli = {
			type = cb_vassalize_mission
			target = BUL
		}
	}
	effect = {
		add_country_modifier = {

			duration = 3650
		}
		hidden_effect = {
			remove_casus_belli = {

	}
