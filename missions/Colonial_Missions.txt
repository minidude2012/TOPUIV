discover_mission = {

	type = empty_provinces
	
	category = DIP
	
	allow = {
		FROM = { has_idea = quest_for_the_new_world }
		range = FROM
		has_port = yes
		NOT = { has_discovered = FROM }
		FROM = {
			is_at_war = no
			num_of_ports = 1
		}
		NOT = {
			FROM = { has_country_modifier = "colonial_enthusiasm" }
		}
		
	}
	abort = {
		OR = {
			FROM = { NOT = { num_of_ports = 1 } }
			FROM = { NOT = { has_idea = quest_for_the_new_world } }
		}
	}
	success = {
		has_discovered = FROM
	}
	chance = {
		factor = 1000
		#tweak for some historical preferences.
		modifier = {
			factor = 0
			FROM = { tag = POR }
			continent = north_america
		}
		modifier = {
			factor = 1.25
			FROM = { tag = POR }
			OR = {
				region = brazil_region
				AND = {
					region = guinea_region
					has_port = yes
				}
			}
		}
		modifier = {
			factor = 0
			FROM = { 
				OR = {
					tag = SPA
					tag = CAS
				}
			}
			OR = {
				continent = north_america
				region = brazil_region
				AND = {
					region = guinea_region
					has_port = yes
				}
			}
		}
		modifier = {
			factor = 1.25
			FROM = { 
				OR = {
					tag = SPA
					tag = CAS
				}
			}
			OR = {
				region = colombia_region
				region = central_america_region
				region = mexico_region
				region = carribeans_region
				region = la_plata_region
			}
		}
		modifier = {
			factor = 1.25
			FROM = {
				OR = {
					tag = ENG
					tag = GBR
					tag = FRA
				}
			}
			continent = north_america
		}
	}
	effect = {
		FROM = {
			add_country_modifier = {
				name = "colonial_enthusiasm"
				duration = 3650
			}
		}
	}
}


build_colony_to_city = {

	type = our_provinces
	
	category = ADM
	
	allow = {
		FROM = { is_at_war = no }
		is_colony = yes
		has_port = yes
		owner = {
			num_of_colonists = 1
			monthly_income = 10
			NOT = {
				any_owned_province = {
					has_province_modifier = local_colonial_expansion
				}
			}
		}
	}
	abort = {
		NOT = { owned_by = FROM }
	}
	success = {
		is_colony = false
		owned_by = FROM
	}
	chance = {
		factor = 1000
		modifier = {
			factor = 1.2
			development = 3
		}		
		modifier = {
			factor = 1.25
			development = 5
		}		
		
	}
	effect = {
		if = {
			limit = { NOT = { has_dlc = "Common Sense" } }
			add_base_tax = 1
			add_base_manpower = 1
			add_base_production = 1
		}
		if = {
			limit = { has_dlc = "Common Sense" }
			add_base_production = 1
			add_permanent_province_modifier = {
				name = "local_colonial_expansion"
				duration = 3650
			}
		}
	}
}

establish_colonial_nation_mission_canada = {

	type = country
	
	category = DIP
	
	target_colonial_region_list = {
		colonial_canada
	}
	
	target_provinces = {
		always = yes
	}
	
	allow = {
		any_owned_province = {
			colonial_region = colonial_canada
			OR = {
				ROOT = { NOT = { religion = catholic } }
				is_religion_grant_colonial_claim = no
				is_religion_grant_colonial_claim = ROOT
			}
			is_city = yes
			is_core = ROOT
		}
		capital_scope = {
			NOT = { continent = north_america }
			NOT = { continent = south_america }
		}
		num_of_ports = 1
		num_of_colonists = 1
		primitives = no
		calc_true_if = {
			all_province = {
				colonial_region = colonial_canada
				range = ROOT
				has_discovered = ROOT
				is_empty = yes
			}
			amount = 10
		}
		NOT = {
			has_country_modifier = "new_colony"
		}
	}
	abort = {
		OR = {
			NOT = { num_of_ports = 1 }
			NOT = { num_of_colonists = 1 }
		}
	}
	success = {
		custom_trigger_tooltip = {
			tooltip = establish_colonial_nation_mission_canada_tooltip
			calc_true_if = {
				amount = 5
				all_target_province = {
					country_or_non_sovereign_subject_holds = ROOT
					is_city = yes
				}
			}
		}
	}
	chance = {
		factor = 4000 #You already have a colony in this colonial region, this should be a priority. Will dissapear by itself if there aren't 10 free provinces in range due to triggers.
		modifier = {
			factor = 2
			NOT = {
				any_subject_country = {
					is_subject_of_type = colony
				}
			}
		}
		modifier = {
			factor = 1.5
			OR = {
				culture_group = scandinavian
				culture_group = british
				culture_group = french
			}
		}
	}
	effect = {
		add_country_modifier = {
			name = "new_colony"
			duration = 3650
		}
	}
}
